#include <iostream>
#include <stdlib.h>

//元素类型
typedef int ElemType ;

//树的节点
struct ANode{
    ElemType data ;
    ANode *left , *right , *parent ;
    int balance ;//height(left) - height(right)
    ANode(){} ;
    ANode(ElemType d){
        data = d ;
        left = NULL ;
        right = NULL ;
        parent = NULL ;
        balance = 0 ;
    }
    void init(ElemType d){
        data = d ;
        left = NULL ;
        right = NULL ;
        parent = NULL ;       
        balance = 0 ;
    }
};
typedef ANode *PANode ;

//空间分配宏，便于实现其他配置方式
#define _AVL_initspace
#define _AVL_allocate() (ANode *)malloc(sizeof(ANode))
#define _AVL_deallocate(p) free(p)

//AVL
class AVLTree{
private:
    PANode root ;
    void _delete_node(PANode p) ;
    PANode _find_subs(PANode pos) ;//寻找删除时用于替换的节点
    PANode _next_node(PANode p) ;//中继前驱
    PANode _prev_node(PANode p) ;//中继后继
    PANode _rotate_left(PANode p ) ;//以p为轴左旋
    PANode _rotate_right(PANode p ) ;//以p为轴右旋
    PANode _balance_left(PANode p ) ;//对p的左枝进行平衡
    PANode _balance_right(PANode p ) ;//对p的右枝进行平衡
    PANode _find_node(PANode p , ElemType d ) ;//从p开始的位置寻找节点
    PANode _find_parent(ElemType d ) ;//寻找d的父节点
    void _in_order(PANode p) ;//p节点的中序遍历
    void _pre_order(PANode p) ;//p节点的前序遍历
    int _check_balance(PANode p) ;//用于测试平衡性是否计算正确
    int _height(PANode p) ;//用于取得子树高度
public:
    ANode bad_node ; //用于获取不存在的节点时给出
    AVLTree() ;//构造空树
    ~AVLTree() ;//销毁树
    PANode Find(ElemType d) ;//查找元素，
    PANode FindMax() ;//查找最大元素，
    PANode FindMin() ;//查找最小元素，
    ElemType& Retrieve(PANode p) ;//获取元素
    PANode Insert(ElemType d) ;//插入元素
    void Delete(ElemType d) ;//按元素删除
    void Erase(PANode p) ;//按节点位置删除
    void Clear() ;//清除整棵树
    void InOrder() ; //中序遍历
    void PreOrder() ; //前序遍历
    static void Sort(ElemType a[] , int n , int d = 1) ;//利用AVL进行排序，静态函数
} ;

//d：1，从小到大，0，从大到小
void AVLTree::Sort(ElemType a[] , int n , int d){
    AVLTree tree ;
    int i ;
    for ( i = 0 ; i < n ; i ++ ){
        tree.Insert(a[i]) ;
    }
    for ( i = 0 ; i < n ; i ++ ){
        PANode p ;
        if ( d ) {
            p = tree.FindMin() ;
        }else{
            p = tree.FindMax() ;   
        }
        a[i] = tree.Retrieve(p) ;
        tree.Erase(p) ;
    }
}

#define max(x,y) ((x)>(y)?(x):(y))

int AVLTree::_height(PANode p ){
    if ( !p ) return -1 ;
    else return max(_height(p->left) , _height(p->right)) + 1 ;
}

#undef max

int AVLTree::_check_balance(PANode p){
    if ( !p ) return 1 ;
    if ( !_check_balance(p->left) || !_check_balance(p->right) )return 0 ;
    else {
        int a = _height(p->left) ;
        int b = _height(p->right) ;
        return a - b == p->balance ;
    }
}

PANode AVLTree::_next_node(PANode p){
    if ( !p ) return NULL ;
    if ( p->right != NULL ){
        p = p->right ;
        while ( p->left ) p = p->left ;
    }else{
        while ( p->parent && p->parent->right == p ) p = p->parent ;
        p = p->parent ;
    }
    return p ;
}

PANode AVLTree::_prev_node(PANode p){
    if ( !p ) return NULL ;
    if ( p->left != NULL ){
        p = p->left ;
        while ( p->right ) p = p->right ;
    }else{
        while ( p->parent && p->parent->left == p ) p = p->parent ;
        p = p->parent ;
    }
    return p ;   
}

PANode AVLTree::Find(ElemType d){   
    return _find_node(root , d) ;
}

PANode AVLTree::FindMax(){
    if ( root == NULL ) return NULL ;
    PANode p = root ;
    while ( p->right ){
        p = p->right ;
    }
    return p ;
}

PANode AVLTree::FindMin(){
    if ( root == NULL ) return NULL ;
    PANode p = root ;
    while ( p->left ){
        p = p->left ;
    }
    return p ;
}

ElemType& AVLTree::Retrieve(PANode p){
    if ( p )
        return p->data ;
    else return bad_node.data ;
}

//左旋
PANode AVLTree::_rotate_left(PANode p ){
    PANode child = p->right ;
    p->right = child->left ;
    if ( child->left ) child->left->parent = p ;
    child->left = p ;
    PANode ancestor = p->parent ;
    if ( ancestor ) {
        if ( ancestor->left == p ) ancestor->left = child ;
        else ancestor->right = child ;
    }else{
        root = child ;
    }
    child->parent = ancestor ;
    p->parent = child ;
    return child ;
}

//右旋
PANode AVLTree::_rotate_right(PANode p ){
    PANode child = p->left ;
    p->left = child->right ;
    if ( child->right ) child->right->parent = p ;
    child->right = p ;
    PANode ancestor = p->parent ;
    if ( ancestor ) {
        if ( ancestor->left == p ) ancestor->left = child ;
        else ancestor->right = child ;
    }else{
        root = child ;
    }
    child->parent = ancestor ;
    p->parent = child ;
    return child ;
}

//左平衡
PANode AVLTree::_balance_left(PANode p ){
    PANode child = p->left ;
    PANode re ;
    int tmp ;
    switch ( child->balance ) {
    case 1:
        re = _rotate_right(p) ;
        re->balance = 0 ;
        re->right->balance = 0 ;
        break;
    case -1:
        _rotate_left(child) ;
        re = _rotate_right(p) ;
        tmp = re->balance ;
        re->balance = 0 ;
        switch(tmp){
        case 0:
            re->left->balance = 0 ;
            re->right->balance = 0 ;
            break;
        case 1:
            re->left->balance = 0 ;
            re->right->balance = -1 ;
            break;
        case -1:
            re->left->balance = 1 ;
            re->right->balance = 0 ;
            break;
        }
        break;
    case 0:
        re = _rotate_right(p) ;
        re->balance = -1 ;
        re->right->balance = 1 ;
        break;
    default:;
        //assert("false") ;
    }
    return re ;
}

//右平衡
PANode AVLTree::_balance_right(PANode p ){
    PANode child = p->right ;
    PANode re ;
    int tmp ;
    switch ( child->balance ) {
    case 1:
        _rotate_right(child) ;
        re = _rotate_left(p) ;
        tmp = re->balance ;
        re->balance = 0 ;
        switch(tmp){
        case 0:
            re->left->balance = 0 ;
            re->right->balance = 0 ;
            break;
        case 1:
            re->left->balance = 0 ;
            re->right->balance = -1 ;
            break;
        case -1:
            re->left->balance = 1 ;
            re->right->balance = 0 ;
            break;
        }
        break;
    case -1:
        re = _rotate_left(p) ;
        re->balance = 0 ;
        re->left->balance = 0 ;
        break;
    case 0:
        re = _rotate_left(p) ;
        re->balance = 1 ;
        re->left->balance = -1 ;
        break;
    default:;
        //assert("false") ;
    }
    return re ;   
}

PANode AVLTree::_find_node(PANode p , ElemType d ){
    if ( !p ) return root ;//说明p是根节点的父节点
    while ( p && p->data != d ){
        if ( d < p->data ) p = p->left ;
        else p = p->right ;
    }   
    return p ;
}

PANode AVLTree::_find_parent(ElemType d ){
    if ( root == NULL ) return NULL ;
    PANode p = NULL ;
    PANode q = root ;
    do{
        if ( q->data == d ) return p ;
        p = q ;
        if ( d < q->data ) q = q->left ;
        else q = q->right ;
    }while (q) ;
    return p ;
}

PANode AVLTree::Insert(ElemType d){   
    //空树
    if ( !root ) {
        root = _AVL_allocate() ;
        root->init(d) ;
        return root ;
    }
   
    PANode parent = _find_parent(d) ;
    PANode child = _find_node(parent , d) ;
    if ( child ){//已经存在该节点
        return child ;
    }

    //不存在该节点
    PANode p = _AVL_allocate() ;
    p->init(d) ;
    p->parent = parent ;
    if ( d < parent->data ) parent->left = p ;
    else parent->right = p ;

    child = p ;
    //查找待调整的树
    while ( parent != NULL ){
        if ( child == parent->left ){//左枝增高
            switch( parent->balance ){
            case 0:
                parent->balance = 1 ;
                child = parent ;
                parent = parent->parent ;
                break;
            case 1:
                parent->balance = 2 ;
                _balance_left(parent) ;
                return p ;
                break;
            case -1:
                parent->balance = 0 ;
                return p ;
                break;
            default:;
                //assert("false") ;
            }
        }else{//右枝增高
            switch( parent->balance ){
            case 0:
                parent->balance = -1 ;
                child = parent ;
                parent = parent->parent ;
                break;
            case 1:
                parent->balance = 0 ;
                return p ;
                break;
            case -1:
                parent->balance = -2 ;
                _balance_right(parent) ;
                return p ;
                break;
            default:;
                //assert("false") ;
            }           
        }
    }
   
    return p ;
}

//该函数仅被Erase过程调用
PANode AVLTree::_find_subs(PANode pos){
    PANode su = pos;
    //ElemType tmp = pos->data ;//可删
    while ( (su->left != NULL) || (su->right != NULL) ) {
        if ( su->balance == 1 ) {
            su = _prev_node(pos) ;
        }else{
            su = _next_node(pos) ;
        }
        pos->data = su->data ;
        pos = su ;
    }
    //su->data = tmp ;//可删，
    return su ;
}

void AVLTree::Erase(PANode pos){
    if ( pos == NULL ) return ;
    PANode su = _find_subs(pos) ;//替换节点
    PANode child = su ;
    PANode parent = child->parent ;
    bool over = false ;
    int tmp ;
    while ( parent && !over ) {
        if ( child == parent->left ){//左枝
            switch (parent->balance){
            case 0:
                parent->balance = -1 ;
                over = true ;
                break;
            case 1:
                parent->balance = 0 ;
                child = parent ;
                parent = parent->parent ;
                break;
            case -1:
                parent->balance = -2 ;
                tmp = parent->right->balance ;
                child = _balance_right(parent) ;
                parent = child->parent ;
                if ( tmp == 0 ) over = true ;
                break;
            default:break;
                //assert("false") ;
            }
        }else{//右枝
            switch (parent->balance){
            case 0:
                parent->balance = 1 ;
                over = true ;
                break;
            case 1:
                parent->balance = 2 ;
                tmp = parent->left->balance ;
                child = _balance_left(parent) ;
                parent = child->parent ;
                if ( tmp == 0 ) over = true ;
                break;
            case -1:
                parent->balance = 0 ;
                child = parent ;
                parent = parent->parent ;
                break;
            default:break;
                //assert("false") ;
            }           
        }
    }   
    if ( su->parent ){
        if (su == su->parent->left) su->parent->left = NULL ;
        else su->parent->right = NULL ;
    }else{
        root = NULL ;
    }
    _AVL_deallocate(su) ;
}

void AVLTree::Delete(ElemType d){
    PANode p = Find(d) ;
    if ( p ) {
        Erase(p) ;
    }
}

void AVLTree::_delete_node(PANode p){
    if ( p != NULL ){
        _delete_node(p->left) ;
        _delete_node(p->right) ;
        _AVL_deallocate(p) ;
    }
}

void AVLTree::Clear(){
    _delete_node(root) ;
    root = NULL ;
}
