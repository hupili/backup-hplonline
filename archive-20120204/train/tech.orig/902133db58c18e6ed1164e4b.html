Treap（随机化的二叉搜索树）
2009-05-30 09:11
(hplonline)2009.5.29

据说Treap就是Tree + Heap的意思。

NOCOW上有很详细的解释：
http://www.nocow.cn/index.php/Treap

强烈推荐这个网站，有很多关于数据结构和算法方面的详细资料。
而且到目前为止非常绿色！非常希望能坚持久点。

核心思想：

BST在极端情况下的效率差的原因在于数据有序。
而在随机插入下BST仍然是高效的。
于是通过给每个元素加入随机产生的优先级，
并维护整棵树在优先级上的堆性质则可。

考虑将一组数据插入Treap完成后。

如果将这时的所有元素按照优先级排序。
按照排后的顺序将这些元素插入一个普通BST。
得到的将和插入Treap的一样。

而优先级是随机给定的，相当于将原来的元素随机排序后插入。
于是可以保证BST在平均情况下的效率。

写法：

我还是在前面写过的搜索树类框架上改的。
要重新写的也就是插入和删除两个过程。
我对优先级维护的是大顶堆，给空节点的优先级赋值为-1。
随机产生的优先级是直接调用rand()，在0..32767之间。

插入：
按照BST的办法找到插入点。插入。
从插入点开始对优先级向上调整成堆。
在堆里面，调整的时候是直接交换节点和他父亲来实现的。
在这里，通过旋转实现，因为旋转可以保证BST的性质不受改变。

删除：
把删除点的优先级视作-1。（这时他的优先级肯定是最小的了）
从删除点对优先级向下调整成堆。
在堆里，调整是通过交换实现。这里依然通过旋转实现“交换”。
调整好后，待删除节点一定是处于叶子的位置了。
于是直接将该节点删除即可。

一点benchmark：

为了测试Treap和普通BST的性能，
选择了20000个已排序点的benchmark。
空间分配用的自己写的space[]的方式。（见下面的空间分配宏处）
先就按照下面的代码跑，9ms
然后做点改动：

//随机宏，便于实现其他随机数产生方式
#define _T_rand() rand()
#define _T_initseed() srand(time(NULL))

中间换个#define _T_rand() 1
这样所有节点的优先级都一样，相当于按照BST方式来搞。

结果为：1735ms

可见引入随机的Treap还是有点用。

测试的时候，如果数据已经洗牌过。则BST比Treap还快。
毕竟BST在实现上少调用了若干rand()，也比较可观。

代码：

#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <time.h>

typedef int ElemType ;

//随机宏，便于实现其他随机数产生方式
#define _T_rand() rand()
#define _T_initseed() srand(time(NULL))

//树的节点
struct TNode{
    ElemType data ;
    TNode *left , *right , *parent ;
    int prior ;
};
typedef TNode *PTNode ;

//空间分配宏，便于实现其他配置方式
#define _T_initspace()
#define _T_allocate() (TNode *)malloc(sizeof(TNode))
#define _T_deallocate(p) free(p)
//TNode space[200000] , *spacehead ;
//#define _T_initspace() spacehead = space
//#define _T_allocate() spacehead++ ;
//#define _T_deallocate(p)

//Treap
//prior成大顶堆
class Treap{
private:
    PTNode root ;
    TNode nil_node ; //空节点
    PTNode nil ;//空节点指针

    void _delete_node(PTNode p) ;
    PTNode _next_node(PTNode p) ;//中继前驱
    PTNode _prev_node(PTNode p) ;//中继后继
    PTNode _rotate_left(PTNode p ) ;//以p为轴左旋
    PTNode _rotate_right(PTNode p ) ;//以p为轴右旋
    PTNode _find_node(PTNode p , ElemType d ) ;//从p开始的位置寻找节点
    PTNode _find_parent(ElemType d ) ;//寻找d的父节点
    void _in_order(PTNode p) ;//p节点的中序遍历
    void _pre_order(PTNode p) ;//p节点的前序遍历
public:
   
    Treap() ;//构造空树
    ~Treap() ;//销毁树
    PTNode Find(ElemType d) ;//查找元素，
    PTNode FindMax() ;//查找最大元素，
    PTNode FindMin() ;//查找最小元素，
    ElemType& Retrieve(PTNode p) ;//获取元素
    PTNode Insert(ElemType d) ;//插入元素
    void Delete(ElemType d) ;//按元素删除
    void Erase(PTNode p) ;//按节点位置删除
    void Clear() ;//清除整棵树
    void InOrder() ; //中序遍历
    void PreOrder() ; //前序遍历
    PTNode Nil() ;//返回空节点，便于调用端判断边界条件
    static void Sort(ElemType a[] , int n , int d = 1) ;//利用T进行排序，静态函数
} ;


//d：1，从小到大，0，从大到小
void Treap::Sort(ElemType a[] , int n , int d){
    Treap tree ;
    int i ;
    for ( i = 0 ; i < n ; i ++ ){
        tree.Insert(a[i]) ;
    //    tree.InOrder() ;
    //    tree.PreOrder() ;
    }
//    tree.InOrder() ;
//    tree.PreOrder() ;
    for ( i = 0 ; i < n ; i ++ ){
        PTNode p ;
        if ( d ) {
            p = tree.FindMin() ;
        }else{
            p = tree.FindMax() ;   
        }
        a[i] = tree.Retrieve(p) ;
        tree.Erase(p) ;
    }
}

Treap::Treap(){
    nil = &nil_node ;
    root = nil ;
    nil_node.left = nil ;
    nil_node.right = nil ;
    nil_node.parent = nil ;
    nil_node.prior = -1 ; //设置为最小，空节点只能在叶子处
}

Treap::~Treap(){
    Clear() ;
}

void Treap::_in_order(PTNode p){
    if ( p != nil ) {
        _in_order(p->left) ;
        std::cout<<p->data<<' ';
        _in_order(p->right) ;
    }
}

void Treap::_pre_order(PTNode p){
    if ( p != nil ) {
        std::cout<<p->data<<' ' ;
        _pre_order(p->left) ;
        _pre_order(p->right) ;
    }   
}

void Treap::InOrder(){
    _in_order(root) ;
    std::cout<<std::endl;
}

void Treap::PreOrder(){
    _pre_order(root) ;
    std::cout<<std::endl;
}

PTNode Treap::_next_node(PTNode p){
    if ( p == nil ) return nil ;
    if ( p->right != nil ){
        p = p->right ;
        while ( p->left != nil ) p = p->left ;
    }else{
        while ( p->parent && p->parent->right == p ) p = p->parent ;
        p = p->parent ;
    }
    return p ;
}

PTNode Treap::_prev_node(PTNode p){
    if ( p == nil ) return nil ;
    if ( p->left != nil ){
        p = p->left ;
        while ( p->right != nil ) p = p->right ;
    }else{
        while ( p->parent && p->parent->left == p ) p = p->parent ;
        p = p->parent ;
    }
    return p ;   
}

PTNode Treap::Find(ElemType d){   
    return _find_node(root , d) ;
}

PTNode Treap::FindMax(){
    if ( root == nil ) return nil ;
    PTNode p = root ;
    while ( p->right != nil ){
        p = p->right ;
    }
    return p ;
}

PTNode Treap::FindMin(){
    if ( root == nil ) return nil ;
    PTNode p = root ;
    while ( p->left != nil ){
        p = p->left ;
    }
    return p ;
}

ElemType& Treap::Retrieve(PTNode p){
    if ( p != nil )
        return p->data ;
    else return nil->data ;
}

//左旋
PTNode Treap::_rotate_left(PTNode p ){
    PTNode child = p->right ;
    p->right = child->left ;
    if ( child->left != nil ) child->left->parent = p ;
    child->left = p ;
    PTNode ancestor = p->parent ;
    if ( ancestor != nil ) {
        if ( ancestor->left == p ) ancestor->left = child ;
        else ancestor->right = child ;
    }else{
        root = child ;
    }
    child->parent = ancestor ;
    p->parent = child ;
    return child ;
}

//右旋
PTNode Treap::_rotate_right(PTNode p ){
    PTNode child = p->left ;
    p->left = child->right ;
    if ( child->right != nil ) child->right->parent = p ;
    child->right = p ;
    PTNode ancestor = p->parent ;
    if ( ancestor != nil ) {
        if ( ancestor->left == p ) ancestor->left = child ;
        else ancestor->right = child ;
    }else{
        root = child ;
    }
    child->parent = ancestor ;
    p->parent = child ;
    return child ;
}

PTNode Treap::_find_node(PTNode p , ElemType d ){
    if ( p == nil ) return root ;//说明p是根节点的父节点
    while ( p != nil && p->data != d ){
        if ( d < p->data ) p = p->left ;
        else p = p->right ;
    }   
    return p ;
}

PTNode Treap::_find_parent(ElemType d ){
    if ( root == nil ) return nil ;
    PTNode p = nil ;
    PTNode q = root ;
    do{
        if ( q->data == d ) return p ;
        p = q ;
        if ( d < q->data ) q = q->left ;
        else q = q->right ;
    }while (q != nil ) ;
    return p ;
}

PTNode Treap::Insert(ElemType d){   
    PTNode parent = _find_parent(d) ;
    PTNode child = _find_node(parent , d) ;

    if ( child != nil ) return child ;
    child = _T_allocate() ;
    child->data = d ;
    child->left = nil ;
    child->right = nil ;
    child->parent = parent ;
    child->prior = _T_rand() ;

    //root为空
    if ( root == nil ) {
        root = child ;
        return root ;
    }

    //将child挂到parent上
    if ( d < parent->data ) parent->left = child ;
    else parent->right = child ;

    //开始按照堆性质调整
    while ( parent != nil && child->prior > parent->prior ){
        if ( child == parent->left )
            child = _rotate_right(parent) ;
        else
            child = _rotate_left(parent) ;
        parent = child->parent ;
    }

    return child ;
}

void Treap::Erase(PTNode pos){
    PTNode left = pos->left ;
    PTNode right = pos->right ;
    while ( left != nil || right != nil ){
        if ( left->prior < right->prior ){
            _rotate_left(pos) ;
        }else{
            _rotate_right(pos) ;
        }
        left = pos->left ;
        right = pos->right ;
    }
    if ( pos == root ){
        root = nil ;
    }else{
        PTNode parent = pos->parent ;
        if ( pos == parent->left ) parent->left = nil ;
        else parent->right = nil ;
    }
    _T_deallocate(pos) ;
}

void Treap::Delete(ElemType d){
    PTNode p = Find(d) ;
    if ( p != nil ) {
        Erase(p) ;
    }
}

void Treap::_delete_node(PTNode p){
    if ( p != nil ){
        _delete_node(p->left) ;
        _delete_node(p->right) ;
        _T_deallocate(p) ;
    }
}

void Treap::Clear(){
    _delete_node(root) ;
    root = nil ;
}
