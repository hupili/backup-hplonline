POJ1141（Brackets Sequence）（DP)

Mar 26th, 2009

(hplonline)2009.3.26

就是给括号序列。
定义了一种regular brackets sequence。
求最少添加多少括号来形成这种序列。

半年前怨念地WA了此题。。今天跑回来，在WA了两盘后，终于搞定了。。。

那么说DP：

s[]是读入的字符串
f[i][j]是i到j的最少的添加数
c[i][j]是对应的添加方案

分几种情况

s[i] == ‘[’ 在右端添 ‘]’ 并求解 s[i + 1][j]
s[i] == ‘(’
s[j] == ‘)’
s[j] == ‘]’类似处理
s[i] == ‘[’ && s[j] == ‘]’ ，求解s[i + 1][j - 1]
s[i] == ‘(’ && s[j] == ‘)’， 类似

除了这些特殊情况外。
对 i <= k < j
分别求解s[i][k] 和 s[k + 1][j]

实现上选取相应的记法来表达策略：
//least bracket num ;
int f[MAXN][MAXN] ;
//strategy:-1,-2,-3,-4,-5
//’[‘,’]’,’(‘,’)’,choose middle
int c[MAXN][MAXN] ;
char s[MAXN] ;

即为负的时候相当于上面的5种特殊情况，为非负的时候就是截成两段的一般情况时的分界点。

于是，程序：

include <stdio.h>

include <string.h>

const int MAXN = 110 ;
const int MAX = 100000000 ;

//least bracket num ;
int f[MAXN][MAXN] ;
//strategy:-1,-2,-3,-4,-5
//’[‘,’]’,’(‘,’)’,choose middle
int c[MAXN][MAXN] ;
char s[MAXN] ;

void output(int l , int r ){
    if ( l > r ) return ;
    switch(c[l][r]){
    case -5 :
        putchar(s[l]);
        output( l + 1 , r - 1 );
        putchar(s[r]);
        break;
    case -1 :
        putchar(‘[‘);
        output( l , r - 1 ) ;
        putchar(‘]’);
        break;
    case -2 :
        putchar(‘[‘);
        output( l + 1 , r ) ;
        putchar(‘]’);
        break;
    case -3 :
        putchar(‘(‘);
        output( l , r - 1 ) ;
        putchar(‘)’);
        break;
    case -4 :
        putchar(‘(‘);
        output( l + 1, r ) ;
        putchar(‘)’);
        break;
    default:
        output( l , c[l][r] ) ;
        output( c[l][r] + 1 , r ) ;
        break;
    }
}

int main(){
    int i , j , k , l ;
    int p ;
   
    gets(s);
    l = strlen(s);
    for ( i = 0 ; i < l ; i ++ ){
        f[i][i] = 1 ;
        switch(s[i]){
        case ‘[‘:c[i][i] = -2 ;break ;
        case ‘]’:c[i][i] = -1 ;break ;
        case ‘(‘:c[i][i] = -4 ;break ;
        case ‘)’:c[i][i] = -3 ;break ;
        }
    }
    for ( i = 1 ; i < l ; i ++ ) f[i][i - 1] = -5 ;

    for ( k = 1 ; k < l ; k ++ ){
        for ( i = 0 ; i < l - k ; i ++ ){
            j = i + k ;
            f[i][j] = MAX ;
            if ( s[i] == ‘(’ && s[j] == ‘)’ || s[i] == ‘[’ && s[j] == ‘]’ ){
                f[i][j] = f[i + 1][j - 1] ;
                c[i][j] = -5 ;
            }
            if ( s[i] == ‘[’ && f[i + 1][j] + 1 < f[i][j] ){
                f[i][j] = f[i + 1][j] + 1 ;
                c[i][j] = -2 ;
            }
            if ( s[j] == ‘]’ && f[i][j - 1] + 1 < f[i][j] ){
                f[i][j] = f[i][j - 1] + 1 ;
                c[i][j] = -1 ;
            }
            if ( s[i] == ‘(’ && f[i + 1][j] + 1 < f[i][j] ){
                f[i][j] = f[i + 1][j] + 1 ;
                c[i][j] = -4 ;
            }
            if ( s[j] == ‘)’ && f[i][j - 1] + 1 < f[i][j] ){
                f[i][j] = f[i][j - 1] + 1 ;
                c[i][j] = -3 ;
            }
            for ( p = i ; p < j ; p ++ ){
                if ( f[i][p] + f[p + 1][j] < f[i][j]){
                    f[i][j] = f[i][p] + f[p + 1][j] ;
                    c[i][j] = p ;
                }
            }
        }   
    }

    output(0,l - 1) ;
    putchar(‘\n’);
   
    return 0 ;
}

最后那句红的putchar是很关键的。。因为是SPJ。。所以没有PE的说法。。
于是我今天又WA了两盘。。
