贴一个BST（BinarySearchTree）

May 11th, 2009

hplonline(2009.5.11)

写过这么多程序了，貌似还没正式写过这个结构。
毕竟以后可能会用到这些东西。
老是小打小闹地遇到特殊问题时写一篇特殊的代码，
编码效率上会很吃亏的，
还是写得像模像样一点，
标准就是自己在若干天后用起来比较方便就行了。

由于学习STL的最后一部分被RBTree卡到了。。
得先学下这个结构。
而他的近亲AVLTree也要学一下。
当然AVL的老爹应该先搞定，于是从BST开始。

我这个版本就有两点稍微可以说的：

1.除了Clear()外没有用递归。
因为像Find，Insert之类的，当前指针p在树上跑的路线都是单边的。
这样大可不必递归。
但Clear() 由于要先清除左右子树，自己写栈来消除递归显然更臃肿，就算了。

2.空间分配方面留了三个宏

define _BST_initspace

define _BST_allocate() (BNode *)malloc(sizeof(BNode))

define _BST_deallocate(p) free(p)

长期做工程的同学可能不理解这个用意。
但是长期刷题的同学应该知道这写猥琐的手脚。。。

比如，刷题的时候换成下面一组：
//BNode space[1000] , *spacehead ;
//#define BST_initspace spacehead = space
//#define BST_allocate() spacehead++ ;
//#define _BST_deallocate(p)

一般来说刷题可以先估计要用的内存大小，
这样搞可以提高速度。
而且把释放内存的宏定义为空。
因为开在全局区的空间，随程序结束自动释放。

总得来说，这一手脚有的时候可以让TLE的变AC。

记得曾经有一道题：
用new和delete超时，
搞了半天，决定不delete了，300ms过了。
把new换成malloc，200ms过了。
换成静态分配，100ms过了。

源码：包含测试部分

include <iostream>

include <stdlib.h>

//元素类型
typedef int ElemType ;

//树的节点
struct BNode{
    ElemType data ;
    BNode left , right , parent ;
    BNode(){
        left = NULL ;
        right = NULL ;
        parent = NULL ;
    }
    BNode(ElemType d){
        data = d ;
        left = NULL ;
        right = NULL ;
        parent = NULL ;
    }
    void init(ElemType d){
        data = d ;
        left = NULL ;
        right = NULL ;
        parent = NULL ;       
    }
};
typedef BNode PBNode ;

//空间分配宏，便于实现其他配置方式

define _BST_initspace

define _BST_allocate() (BNode *)malloc(sizeof(BNode))

define _BST_deallocate(p) free(p)

//BNode space[1000] , *spacehead ;
//#define BST_initspace spacehead = space
//#define BST_allocate() spacehead++ ;
//#define BST_deallocate(p)

//BST
class BinarySearchTree{
private:
    PBNode root ;
    void delete_node(PBNode p) ;
    PBNode find_subs(PBNode pos) ;//寻找删除时用于替换的节点
public:
    BinarySearchTree() ;
    PBNode Find(ElemType d) ;
    PBNode FindMax() ;
    PBNode FindMin() ;
    ElemType& Retrieve(PBNode p) ;
    PBNode Insert(ElemType d) ;
    void Delete(ElemType d) ;//按元素删除
    void Erase(PBNode p) ;//按节点位置删除
    void Clear() ;
    static void Sort(ElemType a[] , int n , int d = 1) ;//利用BST进行排序，静态函数
} ;

//d：1，从小到大，0，从大到小
void BinarySearchTree::Sort(ElemType a[] , int n , int d){
    BinarySearchTree tree ;
    int i ;
    for ( i = 0 ; i <n ; i ++ ){
        tree.Insert(a[i]) ;
    }
    for ( i = 0 ; i < n ; i ++ ){
        PBNode p ;
        if ( d ) {
            p = tree.FindMin() ;
        }else{
            p = tree.FindMax() ;   
        }
        a[i] = tree.Retrieve(p) ;
        tree.Erase(p) ;
    }
}

BinarySearchTree::BinarySearchTree(){
    root = NULL ;
}

PBNode BinarySearchTree::Find(ElemType d){
    PBNode p = root ;
    while ( p && p->data != d ){
        if ( d < p->data ) p = p->left ;
        else p = p->right ;
    }   
    return p ;
}

PBNode BinarySearchTree::FindMax(){
    if ( root == NULL ) return NULL ;
    PBNode p = root ;
    while ( p->right ){
        p = p->right ;
    }
    return p ;
}

PBNode BinarySearchTree::FindMin(){
    if ( root == NULL ) return NULL ;
    PBNode p = root ;
    while ( p->left ){
        p = p->left ;
    }
    return p ;
}

ElemType& BinarySearchTree::Retrieve(PBNode p){
    return p->data ;
}

PBNode BinarySearchTree::Insert(ElemType d){
    //根为空特殊处理
    if ( root == NULL ) {
        root = BST_allocate() ;
        root->init(d) ;       
        return root ;
    }
    PBNode p = root ;
    PBNode q = NULL ;
    while ( p && p->data != d ){
        q = p ;
        if ( d < p->data ) p = p->left ;
        else p = p->right ;
    }
    //该节点存在,直接返回
    if ( p ) {
        return p ;
    }
    //不存在,q为插入点
    p = BST_allocate() ;
    p->init(d) ;
    p->parent = q ;
    if ( d < q->data ){
        q->left = p ;
    }else{
        q->right = p ;
    }
    return p ;
}

PBNode BinarySearchTree::find_subs(PBNode pos){
    PBNode su ;
    if ( pos->left == NULL || pos->right == NULL ) {
        //有一个子树为空
        if ( pos->left == NULL ){
            su = pos->right ;
        }else{
            su = pos->left ;
        }       
    }else{
        //两个均不为空，寻找前驱，即左子树的最右枝
        su = pos->left ;
        while ( su->right ) su = su->right ;
    }
    return su ;
}

void BinarySearchTree::Erase(PBNode pos){
    if ( pos == NULL ) return ;
    PBNode su = find_subs(pos) ;//替换节点
    //根节点
    if ( pos->parent == NULL ) {
        if ( su ){
            PBNode sup = su->parent ;
            if ( su == sup->left ) {
                sup->left = NULL ;
            }else{
                sup->right = NULL ;
            }
            su->parent = NULL ;
        }
        root = su ;
        BST_deallocate(pos) ;
        return ;
    }

    PBNode pa = pos->parent ;//父节点
    //替换
    if ( pos == pa->left ) {
        pa->left = su ;
    }else{
        pa->right = su ;
    }
    if ( su ) {
        PBNode sup = su->parent ;
        if ( su == sup->left ) {
            sup->left = NULL ;
        }else{
            sup->right = NULL ;
        }
        su->parent = pa ;
    }
    BST_deallocate(pos) ;
}

void BinarySearchTree::Delete(ElemType d){
    Erase(Find(d)) ;
}

void BinarySearchTree::delete_node(PBNode p){
    if ( p != NULL ){
        delete_node(p->left) ;
        delete_node(p->right) ;
        BST_deallocate(p) ;
    }
}

void BinarySearchTree::Clear(){
    delete_node(root) ;
    root = NULL ;
}
