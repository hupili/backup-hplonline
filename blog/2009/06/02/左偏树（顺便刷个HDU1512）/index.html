左偏树（顺便刷个HDU1512）

Jun 2nd, 2009

(hplonline)2009.6.2

关于左偏树的资料，
网上到处转载的大多是05年黄源河的论文
《左偏树》
应该说从理论到实现讲解得很清楚了，分享出来。

主要在网上搜的时候，很多地方提到HDU1512
（其实就是一个人提到，到家转出来的）
于是顺便去刷了下。

处理猴子认识与否，是典型的并查集擅长解决的。
每次要得到strongness最大的。
（个人认为应该是strength。。。）
且打完一架之后要合并，于是需要可并堆来支持。
左偏树是性价比很高的可并堆的实现方案。

写法上，和下面的左偏树的实现有点出入：HDU1512.txt
意思上是一样的。

个人的理解：

merge操作作为左偏树的基础。
按照堆的性质，取出待合并的树中最小的作为根即可。
由于限制了右树的“距离”更小。
所以确定根之后，把另一棵树和根的右枝合并。
可以尽可能地减少合并的次数。

距离为0的点的右树是空的，这时候merge可以结束。
而从某节点走到距离为0的节点，只需要他的距离步。
这样可以保证很高的效率。

我的左偏树代码：

include <stdio.h>

include <stdlib.h>

include <iostream>

typedef int ElemType ;

typedef struct leftistnode_type{
    ElemType data ;
    int dis ;
    leftistnode_type left , right ;
}LNode ;

typedef LNode* PLNode ;

//空间分配宏，便于实现其他配置方式

define _L_initspace

define _L_allocate() (LNode *)malloc(sizeof(LNode))

define _L_deallocate(p) free(p)

//ANode space[1000] , *spacehead ;
//#define L_initspace spacehead = space
//#define L_allocate() spacehead++ ;
//#define L_deallocate(p)

class LeftistTree{
private:
    PLNode root ;
    LNode nil_node ;
    PLNode nil ;
    PLNode merge(PLNode t1 , PLNode t2) ;
    void delete_node(PLNode p) ;
public:
    LeftistTree() ;
    ~LeftistTree() ;
    ElemType Root() ;//获得根节点
    void Join(LeftistTree &t) ;//合并另一个树到该树上
    void Insert(ElemType d) ;//插入
    void DeleteRoot() ;//删除
    void Clear() ;//清空树
    bool Empty() ;//判空
    friend LeftistTree ;
} ;

PLNode LeftistTree::merge(PLNode t1 , PLNode t2){
    if ( t1 == nil ) return t2 ;
    if ( t2 == nil ) return t1 ;
    PLNode t ;
    if ( t1->data > t2->data ) {
        t = t1 ; t1 = t2 ; t2 = t ;
    }
    t = merge(t1->right , t2) ;
    if ( t->dis > t1->left->dis ) {
        t1->right = t1->left ;
        t1->left = t ;
    }else{
        t1->right = t ;
    }
    t1->dis = t1->right->dis + 1 ;
    return t1 ;
}

LeftistTree::LeftistTree(){
    nil = &nil_node ;
    nil->left = nil ;
    nil->right = nil ;
    nil->dis = -1 ;//为维护左偏性质而规定
    root = nil ;
}

LeftistTree::~LeftistTree(){
    Clear() ;
}

void LeftistTree::delete_node(PLNode p){
    if ( p != nil ){
        delete_node(p->left) ;
        delete_node(p->right) ;
        L_deallocate(p) ;
    }
}

void LeftistTree::Clear(){
    delete_node(root) ;
}

ElemType LeftistTree::Root(){
    return root->data ;
}

void LeftistTree::Insert(ElemType d){
    PLNode p = L_allocate() ;
    p->data = d ;
    p->left = nil ;
    p->right = nil ;
    p->dis = 0 ;
    root = merge(root , p) ;
}

void LeftistTree::DeleteRoot(){
    PLNode p = root ;
    root = merge(root->left , root->right) ;
    L_deallocate(p) ;
}

void LeftistTree::Join(LeftistTree &t){
    root = _merge(root , t.root) ;
    t.root = t.nil ;
}

bool LeftistTree::Empty(){
    return root == nil ;
}

using namespace std ;

int main(){
    int a[] = {3 , 2 , 9 , 10 , 5 , 4 , 6 , 7} ;
    int sz = sizeof(a) / sizeof(a[0]) ;
    LeftistTree lt ;
    int i ;
    for ( i = 0 ; i < sz ; i ++ )
        lt.Insert(a[i]) ;
    for ( i = 0 ; i < sz ; i ++ ){
        cout<<lt.Root() <<endl;
        lt.DeleteRoot() ;
    }
    return 0 ;
}
