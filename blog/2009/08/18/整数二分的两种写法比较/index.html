整数二分的两种写法比较

Aug 18th, 2009

(hplonline)2009.8.18

看了《C陷阱与缺陷》的ex3-3。
感觉后面的题解貌似过于啰嗦，
并且写法并不优美，
故也实现了一个来比较一下。

程序：

#include <stdio.h>
#include <time.h>
#include <stdlib.h>

const int MAXN = 10000000 ;

int *bsearch1(int *t , int n , int x){
    int *lo = t , *hi = t + n ;
    while ( lo < hi ){
        int *mid = lo + ((hi - lo) >> 1) ;
        if (x < *mid)
            hi = mid ;
        else if (x > *mid)
            lo = mid + 1 ;
        else
            return mid ;
    }
    return NULL ;
}

int *bsearch2(int *t , int n , int x){
    while ( n > 1 ){
        int half = n >> 1 ;
        int *mid = t + half ;
        if ( x >= *mid ){
            t = mid ;
            n -= half ;
        }else {
            n = half ;
        }
    }
    if ( *t == x ) return t ;
    else return NULL ;
}

int *bsearch3(int *t , int n , int x){
    __asm{
        mov eax , t ;
        mov esi , n ;
        mov ecx , x ;

__loop_begin:
        cmp esi , 1 ;
        jng __loop_end ;
        mov ebx , esi ;
        shr ebx , 1 ;
        lea edx , [eax + ebx] ;
        cmp ecx , [edx] ;
        jl _less_than ;
        mov eax , edx ;
        sub esi , ebx ;
        jmp __loop_begin ;
_less_than:
        mov esi , ebx ;
        jmp __loop_begin ;
__loop_end:
       
        cmp [eax] , ecx ;
        je _return ;
        xor eax , eax ;
_return:
        mov t , eax ;
    }
    return t ;
}

int a[MAXN] = {1 , 5 , 7 , 10 , 30 , 32 , 90 , 100} ;
int n = 8 ;

void init(){
    int i ;
    n = MAXN ;
    a[0] = 1 ;
    for ( i = 1 ; i < n ; i ++ ){
        a[i] = a[i - 1] + rand() % 10 + 1 ;
    }


}

void test(int* (*func)(int* , int , int) ){
    int t = clock() ;
   
    int i ;
    for ( i = 0 ; i < n ; i ++ ){
        (*func)(a , n , a[i]) ;
    }

    printf(“%d\n” , clock() - t) ;
}

int main()
{
    srand(time(NULL)) ;
/*    int *p ;
    p = bsearch2(a , n , 30) ;
    if ( p ) printf(“%d\n” , *p) ;
    else printf(“NULL\n”) ;
    return 0 ;*/
    init() ;
    test(bsearch3) ;
    test(bsearch2) ;
    test(bsearch1) ;
    return 0 ;
}


其中bsearch1是作者给出的最终版本。
bsearch2是我看书的时候写下的版本。
bsearch3是上机测试的时候，
弄的一个汇编版本来做比较。

做了两组测试：

由于二分的效率是对数级的，
所以要把数组开到很大，
才能看到之间的差异，
否则，进出函数的开销都会大于这个时间。

1.测试查找到的时候的效率。

vc6.release
MAXN=10000000

随机数产生：
a[i] = a[i - 1] + rand() % 10 + 1 ;
查找：
a[i]，保证每次都能找到，
并且累加了查找每个位置的时间。

结果：
bsearch3:1380
bsearch2:1217
bsearch1:914

2.查找不到时候的效率

vc6.release
MAXN=10000000

随机产生：
a[i] = a[i - 1] + rand() % 10 + 2 ;
查找：
a[i]+1，保证每次都查找不到，
并且累加了各次的时间。

bsearch3:1401
bsearch2:1271
bsearch1:1468

我的汇编版本的表现比较有波动，
但是bsearch2的表现在各次测试中比较稳定，
均比作者给出的版本要好。

关于写法的说明：

作者的写法实际上是市面上经常见到的“三段式”。
就是要判断><=三种情况。
很明显的，代码中出现了两次比较。
我上面的测试是均匀地测了数字出现的各个位置的。
如果有倾向性地测试某一边，
将可以造成在很多时候都进行两次比较的情况。
而条件跳转的开销是很大的。
可以参考这个实验的结果。

我的写法也并不是什么原创的东西。
思路还是学的STL的那一套。
首先是左闭右开的区间表示法。
然后是转化成<起点，长度>的表示法。
这种表示在实现很多事情的时候很方便。
说不到特别明白，也没啥具体的例子，
写了若干回后就感觉到了。

然后，最有影响的地方其实是“两段式”的写法。
这样就只用一次比较。
如果现在的应用比较的不是int的数据，
是个结构体，或者类。
每一次比较都要调用一个函数，
且很费时间。
那么这减少的一次比较将变得特别明显。

但是“两段式”写法有个最大的问题，
就是很多时候掌握不好该怎么写，
最后跑成个TLE，或者WA。
没有在OJ上大量刷题经验的人，
一般都会感觉良好，二分嘛，不就那个意思嘛。。
而这个世界上有很多事情就那个意思，
一动手就不是那个意思了。
我是直到去年刷挂了若干题之后，
才开始仔细想想各种情况下应该怎么写，
使得既准确，又可以很快地实现出来。

可以考虑的问题：

1.用左开右闭的方式怎么写？

2.假设现在要完成的是这样的任务：
返回的指针*i，表示的是把x插入到i的后面一个，原序列依然有序。

返回i的下界怎么写？返回i的上界怎么写？

比如：1 5 7 10 11 11 20
x=11
（下标从0开始）
i的下界是3，上界是5。
