按CIDR方式聚合的中国路由

Oct 4th, 2009

(hplonline)2009.10.4

（重要提示，UESTC.SCIE.NE.07的人如果无意搜到此页，
麻烦不要直接使用，或者你们注明出处）

预处理：

首先，要找到数据，来自APNIC的ftp服务器：
http://ftp.apnic.net/stats/apnic/

这里面有按照各年记录的地址，
assigned-apnic-latest就是最新的。

既然是统计我们国家的，对数据进行一下预处理。
这个处理是很简单的，用excel就行了。
按照|分割数据导入，用自动筛选，国家一栏为CN的取出来就搞定。

然后手上就是规整的跟我们国家有关的数据，类似这样：

apnic|CN|ipv4|58.14.0.0|131072|20050224|allocated
apnic|CN|ipv4|58.16.0.0|65536|20050125|allocated
apnic|CN|ipv4|58.17.0.0|32768|20050201|allocated
apnic|CN|ipv4|58.17.128.0|32768|20050215|allocated
apnic|CN|ipv4|58.18.0.0|65536|20050301|allocated

CIDR聚合方法：

上面的数据，主要关注的是4、5两栏。
58.14.0.0|131072
前面是分配得到的IP地址的起始，
后面是从该地址开始的地址个数。
个数全部是按照2的方幂给出的，
也就是说对该数字取2为底的对数，
就得到了主机号所占的位数。
32减之就得到了网络号的位数。
上面这些网段用这种算法可以看做：
58.14.0.0/15
58.16.0.0/16
58.17.0.0/17
58.17.128.0/17
58.18.0.0/16

要聚合的两条路由一定是相同的网络前缀长度L，
他们网络地址的前L-1位都相同，
第L位是一条为1，一条为0。
这样，我们就可以合并出一条新的网络前缀为L-1的路由，
其网络地址就是较小的那一个（第L位为0的那个）。

比如：
58.17.0.0/17
58.17.128.0/17
可以聚合为：
58.17.0.0/16

这个过程应该不断重复，直到没有可以合并的为止，
像上面这些网段，还可以继续合并。
58.16.0.0/16
58.17.0.0/16
==>58.16.0.0/15

程序：

由于数据文件中的有些列是不用的，
采用scanf的字符集合方法可以比较方便地找到”|”，
具体情况见：scanf的三点用法。

IP地址按照点分十进制读入，
然后处理成一个unsigned long 型，
这样，在比较前缀的时候，
只需要与相关的网络掩码相与就行了。

比较戳瓢的是==的优先级比&高，
忘打括号了，调了半天。。

#include <stdio.h>
#include <list>

using namespace std ;

typedef unsigned long ULONG ;
typedef unsigned char UCHAR ;

typedef struct addr_type{
    ULONG ip ;
    int suffix ;
}ADDR ;

list<ADDR> addrs[34] ;
ULONG mask[33] ;

void init_mask(){
    int i ;
    ULONG t = ~0 ;
    for ( i = 0 ; i <= 32 ; i ++ ){
        mask[i] = ~t ;
        t >>= 1 ;
    }
}

int main()
{
    init_mask() ;
    ADDR tmp ;
    char buf[500] ;
    ULONG a[4] ;
    int hosts ;
    int origin_ip = 0 ;
    int new_ip = 0 ;

    freopen(“CN.txt” , “r” , stdin) ;
    freopen(“out.txt” , “w” , stdout) ;

    while (scanf(“%[^|]” , buf) == 1) {
        getchar() ;//puts(buf) ;
        scanf(“%[^|]” , buf) ;getchar() ;//puts(buf) ;
        scanf(“%[^|]” , buf) ;getchar() ;//puts(buf) ;
        scanf(“%[^|]” , buf) ;getchar() ;//puts(buf) ;
        sscanf(buf , “%d.%d.%d.%d” , a , a + 1 , a + 2 , a + 3) ;
        tmp.ip = ((a[0] * 256 + a[1]) * 256 + a[2]) * 256 + a[3] ;
        scanf(“%[^|]” , buf) ;getchar() ;//puts(buf) ;
        sscanf(buf , “%d” , &hosts) ;
        tmp.suffix = 0 ;
        while ( hosts != 1 ){
            hosts >>= 1 ;
            tmp.suffix ++ ;
        }
        gets(buf) ;       
        addrs[tmp.suffix].push_back(tmp) ;
        origin_ip ++ ;
    }

    int i ;
    bool found ;
    for ( i = 1 ; i <= 32 ; i ++ ){
        ULONG tmp_mask = mask[32 - i - 1] ;
        found = true ;
        while ( found ){
            found = false ;
            list<ADDR>::iterator it1,it2 ;
            for ( it1 = addrs[i].begin() ; it1 != addrs[i].end() ; it1 ++ ){
                it2 = it1 ;
                it2 ++ ;
                for ( ; it2 != addrs[i].end() ; it2 ++ ){
                    if ( ((*it2).ip & tmp_mask) == ((*it1).ip & tmp_mask) ){
                        //== is prior to & !!!
                        found = true ;
                        break ;
                    }
                }
                if ( found ) break ;
            }
            if ( found ){
                tmp.ip = (*it1).ip & tmp_mask ;
                tmp.suffix = (*it1).suffix + 1 ;
                addrs[i + 1].push_back(tmp) ;
                addrs[i].erase(it1) ;
                addrs[i].erase(it2) ;
            }
        }
    }
    for ( i = 0 ; i < 32 ; i ++ ){
        list<ADDR>::iterator it ;
        for ( it = addrs[i].begin() ; it != addrs[i].end() ; it ++ ){
            printf(“%d.%d.%d.%d/%d\n” ,
                UCHAR((*it).ip >> 24) ,
                UCHAR((*it).ip >> 16) ,
                UCHAR((*it).ip >> 8) ,
                UCHAR((*it).ip) ,
                32 - (*it).suffix) ;
            new_ip ++ ;
        }
    }
    printf(“%d,%d” , origin_ip , new_ip) ;
    return 0;
}
